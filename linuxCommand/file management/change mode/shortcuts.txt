[Desktop Entry]
Encoding=UTF-8
Name=application 	//快捷方式名称
Comment=application	//备注
Exec=~/address 		//执行指令的绝对路径
Icon=~/app.png	 	//快捷方式icon的绝对路径
Terminal=false 		//打开软件的时候是否代开终端
StartupNotify=true	//打开通知
Type=Application	//类型
Categories=Application 	//分类
